CREATE TABLE Usuario(id integer PRIMARY KEY, nombre varchar(30), sexo varchar(9), edad integer, dirección varchar(30), correo varchar(30))

CREATE TABLE TiendaProductos(id integer PRIMARY KEY, varchar(30), nombre varchar(30), ubicacion varchar(30), teléfono integer, correo varchar(30), rubro varchar(15)) 

CREATE TABLE TiendaServicios(id integer PRIMARY KEY, varchar(30), nombre varchar(30), ubicacion varchar(30), teléfono integer, correo varchar(30), horario time, rubro varchar(15))

CREATE TABLE Producto(id integer PRIMARY KEY, nombre varchar(30), precio integer, descripción varchar(30))

CREATE TABLE Servicio(id integer PRIMARY KEY, nombre varchar(30), precio integer, descripción varchar(30), fecha_limite date)

CREATE TABLE Compra(id integer, id_usuario integer, id_review varchar(30)
